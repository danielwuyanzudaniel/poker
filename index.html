`); activePlayers.forEach(p => { const fullHand = [...p.hand, ...this.game.board].map(c => this.parseCard(c)); p.finalScore = this.calculateHandRank(fullHand); this.addLog(`🃏 ${p.name.split('|')[1] || p.name}: [${p.hand.join(' ')}] - ${p.finalScore.name}`); }); let messages = []; let potNumber = 1; let playersWithInvest = this.players.filter(p => p.investedThisHand > 0); let loopGuard = 0; while (playersWithInvest.length > 0 && loopGuard < 10) { loopGuard++; let activeInvestments = activePlayers.filter(p => p.investedThisHand > 0.01); if (activeInvestments.length === 0) break; let minInvest = Math.min(...activeInvestments.map(p => p.investedThisHand)); let layerPot = 0; let eligiblePlayers = []; this.players.forEach(p => { if (p.investedThisHand > 0.01) { let extract = Math.min(p.investedThisHand, minInvest); p.investedThisHand -= extract; layerPot += extract; if (activePlayers.includes(p) && Math.abs(extract - minInvest) < 0.01) eligiblePlayers.push(p); } }); layerPot = parseFloat(layerPot.toFixed(2)); if (eligiblePlayers.length === 1) { let w = eligiblePlayers[0]; w.chips = parseFloat((w.chips + layerPot).toFixed(2)); w.winAmount = parseFloat(((w.winAmount || 0) + layerPot).toFixed(2)); let m = `💸 ${w.name.split('|')[1] || w.name} 回收多余底池 $${layerPot}`; messages.push(m); this.addLog(m); this.markWinnerSafe(w); // 赢了边池也算坐下！ } else if (eligiblePlayers.length > 1) { let bestScore = -1; let winners = []; eligiblePlayers.forEach(p => { if (p.finalScore.value > bestScore) { bestScore = p.finalScore.value; winners = [p]; } else if (p.finalScore.value === bestScore) { winners.push(p); } }); let splitAmt = parseFloat((layerPot / winners.length).toFixed(2)); let winnerNames = []; winners.forEach(w => { w.chips = parseFloat((w.chips + splitAmt).toFixed(2)); w.winAmount = parseFloat(((w.winAmount || 0) + splitAmt).toFixed(2)); winnerNames.push(`${w.name.split('|')[1] || w.name}`); this.markWinnerSafe(w); // 平分底池也算坐下！ }); const potName = potNumber === 1 ?
